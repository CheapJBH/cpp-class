```cpp
// 0-1 背包
for (int i = 1; i <= n; i++)

    for (int j = m; j >= v[i]; j--)
    {
        f[j] = max(f[j], f[j - v[i]] + c[i]);
```

### 区间 DP

```cpp
for (int len = 2; len <= n; len++)
```

### 最长上升子序列

```cpp
// O(n log n)
for (int i = 1; i <= n; i++)

    if (vec.empty() || a[i] > vec.back())

    else

}
```

## 图论

### DFS

```cpp
bool vis[maxn];

void dfs(int u)
{
    if (vis[u]) return;
    vis[u] = true;
    for (int v : vec[u])
    {
        dfs(v);
    }
}
```

### BFS

```cpp
int dis[maxn];
memset(dis, -1, sizeof(dis));
queue<int> q;
q.push(1);
dis[1] = 0;

while (!q.empty())
{
    int u = q.front(); q.pop();
    for (int v : vec[u])
    {
        if (dis[v] != -1) continue;
        dis[v] = dis[u] + 1;
        q.push(v);
    }
}
```

```cpp
f[0] = 1, f[1] = 0, f[2] = 0;
```

### B - 多边形（CSP-J 2025）

> 给定 n 根木棍长度，求能组成多边形的子集数，模 998244353。

```cpp
sort(a + 1, a + n + 1);
dp[0] = 1;
int cur = 0;
for (int i = 1; i <= n; i++)
{
    for (int j = cur; j >= 0; j--)
    {
        dp[j + a[i]] = (dp[j + a[i]] + dp[j]) % MOD;
    }
    for (int j = 0; j <= a[i]; j++)
    {
        ans = (ans + dp[j]) % MOD;
    }
    cur += a[i];
}
```
